let tangler = new Tangler(document.querySelector('.joke'), "Initial value (String, number, ...)")
document.querySelector('.rd').addEventListener('click', async()=>{
  let response = await fetch('http://api.icndb.com/jokes/random')
  let data = await response.json()
  tangler.value = data.value.joke
})

tangler.value = ""
